  <div class="quiz-container">

    <h2 id="question">Loading question...</h2>

    <div class="options" id="options"></div>

    <div class="explanation" id="explanation"></div>

    <button class="next-btn" id="nextBtn" style="display:none;">Next Question</button>

    <div class="score" id="score"></div>

  </div>

 

  <script>

    let currentQuestionIndex = 0;

    let score = 0;

    let quizData = [];

 

    // Shuffle helper

    function shuffleArray(arr) {

      for (let i = arr.length - 1; i > 0; i--) {

        const j = Math.floor(Math.random() * (i + 1));

        [arr[i], arr[j]] = [arr[j], arr[i]];

      }

      return arr;

    }

 

    // Load questions from external JSON

    fetch("questions.json")

      .then(response => response.json())

      .then(data => {

        quizData = shuffleArray(data); // shuffle questions

        showQuestion();

        updateProgress();

      })

      .catch(error => {

        document.getElementById("question").innerText = "Error loading quiz data!";

        console.error(error);

      });

 

    function showQuestion() {

      const questionObj = quizData[currentQuestionIndex];

      document.getElementById("question").innerText = questionObj.question;

 

      const optionsContainer = document.getElementById("options");

      optionsContainer.innerHTML = "";

 

      // shuffle options each time

      const optionIndexes = questionObj.options.map((_, i) => i);

      shuffleArray(optionIndexes);

 

      optionIndexes.forEach(idx => {

        const button = document.createElement("button");

        button.innerText = questionObj.options[idx];

        button.onclick = () => checkAnswer(button, idx);

        optionsContainer.appendChild(button);

      });

 

      document.getElementById("explanation").style.display = "none";

      document.getElementById("nextBtn").style.display = "none";

    }

 

    function checkAnswer(button, index) {

      const questionObj = quizData[currentQuestionIndex];

      const buttons = document.querySelectorAll(".options button");

 

      buttons.forEach(btn => btn.disabled = true);

 

      if (index === questionObj.answer) {

        button.classList.add("correct");

        score++;

      } else {

        button.classList.add("wrong");

        buttons.forEach((btn, i) => {

          if (i === questionObj.answer) btn.classList.add("correct");

        });

      }

 

      const explanationDiv = document.getElementById("explanation");

      explanationDiv.innerText = "Explanation: " + questionObj.explanation;

      explanationDiv.style.display = "block";

 

      document.getElementById("nextBtn").style.display = "inline-block";

    }

 

    document.getElementById("nextBtn").onclick = () => {

      currentQuestionIndex++;

      if (currentQuestionIndex < quizData.length) {

        showQuestion();

        updateProgress();

      } else {

        showScore();

      }

    };

 

    function updateProgress() {

      const progress = ((currentQuestionIndex) / quizData.length) * 100;

      document.getElementById("progressBar").style.width = progress + "%";

    }

 

    function showScore() {

      document.querySelector(".quiz-container").innerHTML =

        `<h2>Quiz Finished!</h2>

         <p class="score">Your score: ${score} / ${quizData.length}</p>`;

    }

  </script>
